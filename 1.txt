import java.io.FileInputStream;
import java.io.FileOutputStream;
import java.io.IOException;

public class ReadWriteBytes {

  public static void main(String[] args) {
    try (FileInputStream inputStream = new FileInputStream("1.txt");
         FileOutputStream outputStream = new FileOutputStream("bytes.txt")) {
      byte[] buffer = new byte[1024]; // Adjust buffer size as needed
      int bytesRead;

      // Read bytes from the input file and write them to the output file
      while ((bytesRead = inputStream.read(buffer)) > 0) {
        outputStream.write(buffer, 0, bytesRead);
      }

      System.out.println("File content successfully copied to bytes.txt");
    } catch (IOException e) {
      System.err.println("Error: " + e.getMessage());
    }
  }
}
